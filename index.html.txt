    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-app.js";
        import { getFirestore, collection, onSnapshot, doc, addDoc, updateDoc, deleteDoc, query, orderBy } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-firestore.js";
        import { getAuth, signInWithEmailAndPassword, signOut, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-auth.js";

        // ==========================================
        // COLE SUAS CREDENCIAIS DO FIREBASE AQUI:
        // ==========================================
        const firebaseConfig = {
            apiKey: "SUA_API_KEY",
            authDomain: "SEU_AUTH_DOMAIN",
            projectId: "SEU_PROJECT_ID",
            storageBucket: "SEU_STORAGE_BUCKET",
            messagingSenderId: "SEU_MESSAGING_SENDER_ID",
            appId: "SEU_APP_ID"
        };

        // Inicializar Firebase
        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);
        const auth = getAuth(app);

        let dadosJogadores = [];
        let usuarioLogado = null;

        // Escuta o banco de dados em TEMPO REAL (Atualiza o cliente sozinho)
        const q = query(collection(db, "jogadores"), orderBy("gols", "desc"));
        onSnapshot(q, (snapshot) => {
            dadosJogadores = [];
            snapshot.forEach((doc) => {
                dadosJogadores.push({ id: doc.id, ...doc.data() });
            });
            renderizarRankingCliente();
            if(usuarioLogado) renderizarPainelAdmin();
        });

        // Monitor de Autenticação do Admin
        onAuthStateChanged(auth, (user) => {
            if (user) {
                usuarioLogado = user;
                document.getElementById('btnAlternarJanela').innerHTML = `<i class="fa-solid fa-chart-simple"></i> Ver Ranking`;
                document.getElementById('btnAlternarJanela').onclick = () => alternarVisualizacao('secao-cliente');
                alternarVisualizacao('secao-admin');
            } else {
                usuarioLogado = null;
                fazerLogoutInterface();
            }
        });

        // FUNÇÕES INTERNAS ATRIBUÍDAS AO ESCOPO GLOBAL PARA COMPATIBILIDADE
        window.abrirLoginAdmin = () => usuarioLogado ? alternarVisualizacao('secao-admin') : abrirModal('modal-login');
        
        window.autenticarAdmin = async () => {
            const email = document.getElementById('login-user').value;
            const pass = document.getElementById('login-pass').value;
            try {
                await signInWithEmailAndPassword(auth, email, pass);
                fecharModal('modal-login');
            } catch (error) {
                alert("Erro ao entrar: Verifique o e-mail e senha cadastrados no Authentication do Firebase.");
            }
        };

        window.fazerLogout = () => signOut(auth);

        function fazerLogoutInterface() {
            document.getElementById('btnAlternarJanela').innerHTML = `<i class="fa-solid fa-lock"></i> Admin`;
            document.getElementById('btnAlternarJanela').onclick = () => abrirLoginAdmin();
            alternarVisualizacao('secao-cliente');
        }

        function renderizarRankingCliente() {
            const container = document.getElementById('ranking-container');
            container.innerHTML = "";
            dadosJogadores.forEach((jogador, index) => {
                const fotoHTML = jogador.foto ? `<img src="${jogador.foto}">` : `<i class="fa-solid fa-user"></i>`;
                const row = document.createElement('div');
                row.className = 'jogador-row';
                row.onclick = () => abrirModalGolsCliente(jogador.id);
                row.innerHTML = `
                    <div class="posicao">${index + 1}º</div>
                    <div class="foto-container">${fotoHTML}</div>
                    <div class="dados">
                        <div class="nome-jogador">${jogador.nome}</div>
                        <div class="equipe-jogador">${jogador.equipe}</div>
                    </div>
                    <div class="gols-contador"><span class="gols-numero">${jogador.gols}</span><span class="gols-label">Gols</span></div>
                `;
                container.appendChild(row);
            });
        }

        function renderizarPainelAdmin() {
            const container = document.getElementById('admin-lista-jogadores');
            container.innerHTML = "";
            dadosJogadores.forEach((jogador, index) => {
                const fotoHTML = jogador.foto ? `<img src="${jogador.foto}">` : `<i class="fa-solid fa-user"></i>`;
                const card = document.createElement('div');
                card.className = 'card-admin';
                card.innerHTML = `
                    <div class="grid-admin-jogador">
                        <div style="display:flex; align-items:center; gap:12px;">
                            <span style="color:var(--accent); font-weight:bold;">#${index+1}</span>
                            <div class="foto-container" style="width:40px; height:40px;">${fotoHTML}</div>
                            <div><strong>${jogador.nome}</strong><span style="display:block;font-size:0.8rem;color:var(--texto-mutado);">${jogador.equipe}</span></div>
                        </div>
                        <div style="display:flex; align-items:center; gap:15px;">
                            <div class="controles-gols">
                                <button class="btn-gol" onclick="modificarGolsDB('${jogador.id}', ${jogador.gols}, -1)"><i class="fa-solid fa-minus"></i></button>
                                <span style="font-weight:bold; min-width:20px; text-align:center;">${jogador.gols}</span>
                                <button class="btn-gol" onclick="modificarGolsDB('${jogador.id}', ${jogador.gols}, 1)"><i class="fa-solid fa-plus"></i></button>
                            </div>
                            <div style="display:flex; gap:5px;">
                                <button class="btn btn-secundario" style="padding:8px;" onclick="abrirModalJogador('${jogador.id}')"><i class="fa-solid fa-pen"></i></button>
                                <button class="btn btn-secundario" style="padding:8px; color:var(--accent);" onclick="abrirGerenciarLinksAdmin('${jogador.id}')"><i class="fa-solid fa-video"></i> (${jogador.links ? jogador.links.length : 0})</button>
                                <button class="btn btn-danger" style="padding:8px;" onclick="excluirJogadorDB('${jogador.id}')"><i class="fa-solid fa-trash"></i></button>
                            </div>
                        </div>
                    </div>
                `;
                container.appendChild(card);
            });
        }

        window.modificarGolsDB = async (id, golsAtuais, valor) => {
            const novaQtd = Math.max(0, golsAtuais + valor);
            await updateDoc(doc(db, "jogadores", id), { gols: novaQtd });
        };

        window.excluirJogadorDB = async (id) => {
            if(confirm("Deseja mesmo remover do ranking?")) await deleteDoc(doc(db, "jogadores", id));
        };

        window.salvarJogador = async () => {
            const idEdit = document.getElementById('jogador-id-edit').value;
            const nome = document.getElementById('form-nome').value;
            const equipe = document.getElementById('form-equipe').value;
            const foto = document.getElementById('form-foto').value;

            if(!nome || !equipe) return alert("Campos obrigatórios!");

            if(idEdit) {
                await updateDoc(doc(db, "jogadores", idEdit), { nome, equipe, foto });
            } else {
                const gols = parseInt(document.getElementById('form-gols').value) || 0;
                await addDoc(collection(db, "jogadores"), { nome, equipe, foto, gols, links: [] });
            }
            fecharModal('modal-jogador');
        };

        window.abrirGerenciarLinksAdmin = (id) => {
            const j = dadosJogadores.find(item => item.id === id);
            if(!j) return;
            document.getElementById('links-jogador-id').value = j.id;
            document.getElementById('links-admin-jogador').innerText = j.nome;
            document.getElementById('novo-link-url').value = "";
            atualizarListaLinksAdmin(j);
            abrirModal('modal-links-admin');
        };

        function atualizarListaLinksAdmin(jogador) {
            const container = document.getElementById('lista-links-admin');
            container.innerHTML = "";
            const links = jogador.links || [];
            links.forEach((link, idx) => {
                const item = document.createElement('div');
                item.className = 'link-item';
                item.innerHTML = `<a href="${link}" target="_blank">${link}</a><button class="btn-gol" onclick="removerLinkDB('${jogador.id}', ${idx})"><i class="fa-solid fa-trash" style="color:var(--danger)"></i></button>`;
                container.appendChild(item);
            });
        }

        window.adicionarLinkExterno = async () => {
            const id = document.getElementById('links-jogador-id').value;
            const url = document.getElementById('novo-link-url').value;
            if(!url) return;
            const j = dadosJogadores.find(item => item.id === id);
            const novosLinks = j.links || [];
            novosLinks.push(url);
            await updateDoc(doc(db, "jogadores", id), { links: novosLinks });
            abrirGerenciarLinksAdmin(id);
        };

        window.removerLinkDB = async (id, index) => {
            const j = dadosJogadores.find(item => item.id === id);
            const novosLinks = j.links || [];
            novosLinks.splice(index, 1);
            await updateDoc(doc(db, "jogadores", id), { links: novosLinks });
            abrirGerenciarLinksAdmin(id);
        };

        function abrirModalGolsCliente(id) {
            const j = dadosJogadores.find(item => item.id === id);
            if(!j) return;
            document.getElementById('video-cliente-nome').innerText = j.nome;
            document.getElementById('video-cliente-equipe').innerText = j.equipe;
            const container = document.getElementById('lista-links-cliente');
            container.innerHTML = "";
            const links = j.links || [];
            if(links.length === 0) container.innerHTML = "<p style='color:var(--texto-mutado)'>Nenhum vídeo salvo.</p>";
            links.forEach((link, idx) => {
                const item = document.createElement('div');
                item.className = 'link-item';
                item.innerHTML = `<a href="${link}" target="_blank"><i class="fa-solid fa-play"></i> Assistir ao gol nº ${idx+1}</a>`;
                container.appendChild(item);
            });
            abrirModal('modal-videos-cliente');
        }

        function alternarVisualizacao(secaoId) {
            document.querySelectorAll('.view-section').forEach(s => s.classList.remove('active'));
            document.getElementById(secaoId).classList.add('active');
        }
    </script>

    <script>
        // Funções estruturais de controle de tela (Modais)
        function abrirModal(id) { document.getElementById(id).style.display = 'flex'; }
        function fecharModal(id) { document.getElementById(id).style.display = 'none'; }
        function abrirModalJogador(id = null) {
            if(id) {
                document.getElementById('modal-jogador-titulo').innerText = "Editar Jogador";
                document.getElementById('jogador-id-edit').value = id;
                document.getElementById('grupo-gols-inicial').style.display = 'none';
            } else {
                document.getElementById('modal-jogador-titulo').innerText = "Novo Artilheiro";
                document.getElementById('jogador-id-edit').value = "";
                document.getElementById('form-nome').value = "";
                document.getElementById('form-equipe').value = "";
                document.getElementById('form-foto').value = "";
                document.getElementById('form-gols').value = "0";
                document.getElementById('grupo-gols-inicial').style.display = 'block';
            }
            abrirModal('modal-jogador');
        }
    </script>
